```
if (/firefox/i.test(navigator.userAgent)) {
            alert("火狐");
        } else if (/chrome/i.test(navigator.userAgent)) {
            alert("chrome");
        } else if (/msie/i.test(navigator.userAgent)) {
            alert("ie9-");
        } else if (ActiveXObject in window) {
            alert("ie");
        }
```
